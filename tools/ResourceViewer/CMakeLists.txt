FIND_PACKAGE(Qt4 REQUIRED)
FIND_PACKAGE(squish REQUIRED)

INCLUDE(${QT_USE_FILE})

SET(RW_HEADERS
    MainWindow.h
    FileModel.h
)

SET(RW_SOURCES
    main.cpp
    MainWindow.cpp
    FileModel.cpp
)

SET(RW_UI
    MainWindow.ui
)

INCLUDE_DIRECTORIES(
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_SOURCE_DIR}/libRanger
  ${SQUISH_INCLUDE_DIR}
)

QT4_WRAP_CPP(RW_MOC_SOURCES ${RW_HEADERS})
QT4_WRAP_UI(RW_UI_SOURCES ${RW_UI})

ADD_EXECUTABLE(opensr-resourceviewer ${RW_SOURCES} ${RW_UI_SOURCES} ${RW_MOC_SOURCES} ${RW_HEADERS})
IF(WIN32)
    SET_TARGET_PROPERTIES(opensr-resourceviewer PROPERTIES COMPILE_FLAGS "/Zc:wchar_t-")
ENDIF(WIN32)
TARGET_LINK_LIBRARIES(opensr-resourceviewer ranger ${QT_LIBRARIES} ${SQUISH_LIBRARIES})

INSTALL(TARGETS opensr-resourceviewer RUNTIME DESTINATION bin)
